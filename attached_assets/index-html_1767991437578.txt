<script>
const cups = document.querySelectorAll('.cup');
const message = document.getElementById('message');
let ballIndex = 0;

// Initial orange placement
function placeBall() {
  cups.forEach((cup, i) => {
    cup.querySelector('.orange').style.display = i === ballIndex ? 'block' : 'none';
  });
}

// Shuffle cups with actual horizontal movement
function shuffleCups(times = 10) {
  const positions = [0, 160, 320]; // x positions of cups
  let i = 0;

  message.textContent = "Try to follow the orange!";

  function step() {
    if (i >= times) return;

    // Pick two cups to swap
    let a = Math.floor(Math.random()*3);
    let b;
    do { b = Math.floor(Math.random()*3); } while(b===a);

    // Animate swapping
    const tempLeft = cups[a].style.left;
    cups[a].style.left = cups[b].style.left;
    cups[b].style.left = tempLeft;

    // Update ball index if necessary
    if (ballIndex === a) ballIndex = b;
    else if (ballIndex === b) ballIndex = a;

    i++;
    setTimeout(step, 400);
  }
  step();
}

// Click event to guess
cups.forEach(cup => {
  cup.addEventListener('click', () => {
    if (cup.dataset.index == ballIndex) {
      message.textContent = "🎉 You found the orange!";
    } else {
      message.textContent = "😢 Wrong cup! Try again.";
    }
    placeBall();
  });
});

// Setup
placeBall();
document.getElementById('shuffleBtn').addEventListener('click', shuffleCups);
</script>
